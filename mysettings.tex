	\begingroup
	\parindent \z@

	\parfillskip -\@pnumwidth
	\leavevmode \bfseries
	\advance \leftskip \@tempdima
	\hskip -\leftskip
	#1\nobreak
	\hfil
	\nobreak
	\hb@xt@ \@pnumwidth {\hss #2\kern -\p@ \kern \p@ }%
	\par
	\endgroup
	\fi
}
